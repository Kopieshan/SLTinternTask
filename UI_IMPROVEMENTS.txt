━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

BEFORE:
  - Basic ElevatedButton
  - Simple icon and text
  - Standard padding

AFTER:
  - Photo button: OutlinedButton (secondary action style)
    - Larger: minimumSize 50px height
    - Icon: Icons.add_a_photo (more specific)
    
  - Submit button: ElevatedButton (primary action)
    - Larger: minimumSize 54px height
    - Icon: Icons.upload_rounded (modern rounded style)
    - Font weight 600 (semi-bold)
    - Full width

  - Retake button: TextButton (tertiary action)
    - Icon: Icons.refresh
    - Smaller, less prominent

WHY: Button hierarchy shows importance. Primary actions are prominent, 
secondary actions are outlined, tertiary are text-only.
